def predict_function(model, df):
    df.loc[:, df.dtypes == 'object'] = df.select_dtypes(['object']).apply(lambda x: x.astype('category'))
    return model.predict_proba(df)[:, 1]

rfc_explainer = dx.Explainer(rfc_model, X_test, y_test, predict_function=predict_function)
